CREATE VIEW SCOTT.aaa
AS
SELECT empno,ENAME,SAL FROM SCOTT.EMP
WHERE SAL BETWEEN 2000 and 5000 AND ENAME LIKE '%A%'


CREATE VIEW SCOTT.ps4
AS
SELECT e.empno,e.ENAME,e.DEPTNO,e.HIREDATE FROM SCOTT.emp e
join scott.DEPT r ON e.DEPTNO=r.DEPTNO
WHERE r.LOC in ('NEW YORK','CHICAGO')
CREATE OR REPLACE VIEW SCOTT.AAA
AS
SELECT empno,ENAME,SAL,hiredate FROM SCOTT.EMP
WHERE SAL BETWEEN 2000 and 5000 AND ENAME LIKE '%A%'
